```powershell
Settings=@(
@{key=$Languages[0];Value=$DisplayNames[0];}
@{key=$Languages[1];Value=$DisplayNames[1];}
@{key=$Languages[2];Value=$DisplayNames[2];})}

@{key=$Languages[0];Value=$Tooltips[0];}

@{key=$Languages[2];Value=$Tooltips[2];})}

```
